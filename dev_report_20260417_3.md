```cpp
// interpreter.h - 元组表达式执行
case claw::ast::Expression::Kind::Tuple: {
    return evaluate_tuple(static_cast<claw::ast::TupleExpr*>(expr));
}

Value evaluate_tuple(claw::ast::TupleExpr* tup) {
    std::vector<Value> elements;
    for (size_t i = 0; i < tup->size(); i++) {
        elements.push_back(evaluate(tup->get_element(i)));
    }
    // Store tuple as RuntimeValue array
    RuntimeValue rv;
    rv.type_name = "tuple";
    rv.size = elements.size();
    for (const auto& elem : elements) {
        RuntimeValue elem_rv;
        elem_rv.scalar = elem;
        elem_rv.size = 1;
        rv.array.push_back(elem_rv);
    }
    runtime.set_var("_tuple_temp", rv);
    return elements.empty() ? Value() : elements[0];
}

// codegen.cpp - 元组代码生成
Value* CodeGenerator::codegen_tuple(ast::TupleExpr* tup) {
    if (tup->size() > 0) {
        return codegen_expression(tup->get_element(0));
    }
    return ConstantInt::get(Type::getInt32Ty(context_), 0);
}
```
